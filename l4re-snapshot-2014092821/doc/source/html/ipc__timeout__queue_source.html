16  * \brief Timeout queue to be used in l4re server loop
17  */

51  /**
52  * \brief Determine if a timeout has happened
53  * \param now the current time.
54  * \return true if there is at least one expired timeout in the queue,
55  * false if not.
56  */
57  bool timeout_expired(l4_kernel_clock_t now) const
58  {
59  if (_timeouts.empty())
60  return false;
61 
62  return _next_timeout <= now;
63  }
64 
65  /**
66  * \brief Get the time for the next timeout.
67  * \return the time for the next timeout or 0 if there is none
68  */
69  l4_kernel_clock_t next_timeout() const
70  {
71  return _next_timeout;
72  }

74  /**
75  * \brief run the callbacks of expired timeouts
76  * \param now the current time.
77  */
78  void handle_expired_timeouts(l4_kernel_clock_t now)
79  {
80  while (!_timeouts.empty())
81  {
82  Queue::Iterator top = _timeouts.begin();
83  if ((*top)->_timeout > now)
84  return;
85 
86  Timeout *t = *top;
87  top = _timeouts.erase(top);
88  t->expired();
89  if (!_timeouts.empty())
90  _next_timeout = (*_timeouts.begin())->timeout();
91  else
92  _next_timeout = 0;
93  }
94  }
95 
96  /**
97  * \brief Add a timeout to the queue
98  * \param timeout timeout object to add
99  * \param time the time when the timeout expires
100  * \pre \a timeout must not be in any queue already
101  */
102  void add(Timeout *timeout, l4_kernel_clock_t time)
103  {
104  timeout->_timeout = time;
105  Queue::Iterator i = _timeouts.begin();
106  while (i != _timeouts.end() && (*i)->timeout() > time)
107  {}
108  _timeouts.insert_before(timeout, i);
109  _next_timeout = (*_timeouts.begin())->timeout();
110  }
111 
112  /**
113  * \brief Remove \a timeout from the queue.
114  * \param timeout timeout to remove from timeout queue
115  * \pre \a timeout must be in this queue
116  */
117  void remove(Timeout *timeout)
118  {
119  _timeouts.remove(timeout);
120  if (_next_timeout == timeout->timeout())
121  {
122  if (!_timeouts.empty())
123  _next_timeout = (*_timeouts.begin())->timeout();
124  else
125  _next_timeout = 0;
126  }
127  }
128 
129 private:
130  typedef cxx::H_list<Timeout> Queue;
131  Queue _timeouts;
132  l4_kernel_clock_t _next_timeout;
133 };

137 /**
138  * \brief Loop hooks mixin for integrating a timeout queue into the server
139  * loop.
140  * \tparam HOOKS has to inherit from Timeout_queue_hooks<> and provide
141  * the functions now() and timeout_br(), whereas now() has to
142  * return the current time and timeout_br() the buffer register
143  * that shall be used for the timeout.
144  */
145 template< typename HOOKS >
146 class Timeout_queue_hooks
147 {
148  l4_kernel_clock_t _now()
149  { return static_cast<HOOKS*>(this)->now(); }
150 
151  unsigned _timeout_br()
152  { return static_cast<HOOKS*>(this)->timeout_br(); }
153 
154 public:
155  l4_timeout_t timeout()
156  {
157  l4_kernel_clock_t t = queue.next_timeout();
158  if (t)
159  return l4_timeout(L4_IPC_TIMEOUT_0, l4_timeout_abs(t, _timeout_br()));
160  return L4_IPC_SEND_TIMEOUT_0;
161  }
162 
163  void setup_wait(L4::Ipc::Istream &, L4::Ipc_svr::Reply_mode mode)
164  {
165  // we must handle the timer only when called after a possible reply
166  // otherwise we probably destroy the reply message.
167  if (mode == L4::Ipc_svr::Reply_separate)
168  {
169  l4_kernel_clock_t now = _now();
170  if (queue.timeout_expired(now))
171  queue.handle_expired_timeouts(now);
172  }
173  }
174 
175  L4::Ipc_svr::Reply_mode before_reply(long, L4::Ipc::Ostream const &)
176  {
177  // split up reply and wait when a timeout has expired
178  if (queue.timeout_expired(_now()))
179  return L4::Ipc_svr::Reply_separate;
180  return L4::Ipc_svr::Reply_compound;
181  }

183  Timeout_queue queue; ///< Use this timeout queue
184 };
