48 /**
49  * \ingroup l4vbus_module
50  * \brief Device on a virtual bus (V-BUS)
51  */
52 class Device : public Pm<Device>
53 {
54 public:
55  Device() : _dev(L4VBUS_NULL) {}
56 
57  Device(L4::Cap<Vbus> bus, l4vbus_device_handle_t dev)
58  : _bus(bus), _dev(dev) {}
59 
60  /**
61  * \brief access the V-BUS capability of the underlying virtual bus.
62  * \return the capability to the underlying V-BUS.
63  */
64  L4::Cap<Vbus> bus_cap() const { return _bus; }
65 
66  /**
67  * \brief access the device handle of this device.
68  * \return the device handle for this device.
69  *
70  * The device handle is used to directly address the device on its virtual
71  * bus.
72  */
73  l4vbus_device_handle_t dev_handle() const { return _dev; }
74 
75 
76  /**
77  * \copydoc l4vbus_get_device_by_hid()
78  * \note \a vbus and \a parent are Device#_bus and Device#_dev respectively.
79  */
80  int device_by_hid(Device *child, char const *hid,
81  int depth = L4VBUS_MAX_DEPTH,
82  l4vbus_device_t *devinfo = 0) const
83  {
84  child->_bus = _bus;
85  return l4vbus_get_device_by_hid(_bus.cap(), _dev, &child->_dev, hid,
86  depth, devinfo);
87  }
88 
89  /**
90  * \copydoc l4vbus_get_next_device()
91  * \note \a vbus and \a parent are Device#_bus and Device#_dev respectively.
92  */
93  int next_device(Device *child, int depth = L4VBUS_MAX_DEPTH,
94  l4vbus_device_t *devinfo = 0) const
95  {
96  child->_bus = _bus;
97  return l4vbus_get_next_device(_bus.cap(), _dev, &child->_dev, depth,
98  devinfo);
99  }
100 
101  /**
102  * \copydoc l4vbus_get_resource()
103  * \note \a vbus and \a dev are Device#_bus and Device#_dev respectively.
104  */
105  int get_resource(int res_idx, l4vbus_resource_t *res) const
106  {
107  return l4vbus_get_resource(_bus.cap(), _dev, res_idx, res);
108  }
109 
110  /**
111  * \copydoc l4vbus_is_compatible()
112  * \note \a vbus and \a dev are Device#_bus and Device#_dev respectively.
113  */
114  int is_compatible(char const *cid) const
115  { return l4vbus_is_compatible(_bus.cap(), _dev, cid); }
116 
117  /**
118  * \brief Test if two devices are the same V-BUS device.
119  * \return true if the two devices are the same, false else.
120  */
121  bool operator == (Device const &o) const
122  {
123  return _bus == o._bus && _dev == o._dev;
124  }
125 
126  /**
127  * \brief Test if two devices are not the same.
128  * \return true if the two devices are different, false else.
129  */
130  bool operator != (Device const &o) const
131  {
132  return _bus != o._bus || _dev != o._dev;
133  }
134 
135 protected:
136  L4::Cap<Vbus> _bus; /*!< The V-BUS capability (where this device is
137  located on). */
138  l4vbus_device_handle_t _dev; ///< The device handle for this device.
139 };
