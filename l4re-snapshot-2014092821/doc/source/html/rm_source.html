40 /**
41  * \defgroup api_l4re_rm Region map API
42  * \ingroup api_l4re
43  * \brief Virtual address-space management.
44  *
45  * The central purpose of the region-map API is to provide means to manage the
46  * virtual memory address space of an L4 task. A region-map object implements
47  * two protocols. The first protocol is the kernel page-fault protocol, to resolve
48  * page faults for threads running in an L4 task. The second protocol is
49  * the region-map protocol itself, that allows to attach a data-space object
50  * to a region of the virtual address space.
51  *
52  * There are two basic concepts provided by a region-map abstraction:
53  * - Regions provide a means to create a view to a data space (or parts of a
54  * data space).
55  * - Areas provide a means to reserve areas in a virtual memory address space
56  * for special purposes. A reserved area is skipped when searching for
57  * an available range of virtual memory, or may be explicitly used to search
58  * only within that area.
59  *
60  * \see \link api_l4re_dataspace Data-Space API. \endlink, L4Re::Dataspace,
61  * L4Re::Rm
62  */
63

163  Auto_region &operator = (Auto_region const &o) throw()
164  {
165  if (&o != this)
166  {
167  if (_rm.is_valid())
168  _rm->detach(l4_addr_t(_addr), 0);
169  _rm = o._rm;
170  _addr = o.release();
171  }
172  return *this;
173  }
174 
175  ~Auto_region() throw()
176  {
177  if (_rm.is_valid())
178  _rm->detach(l4_addr_t(_addr), 0);
179  }
180 
181  T get() const throw() { return _addr; }
182  T release() const throw() { _rm = L4::Cap<Rm>::Invalid; return _addr; }
183  void reset(T addr, L4::Cap<Rm> const &rm) throw()
184  {
185  if (_rm.is_valid())
186  _rm->detach(l4_addr_t(_addr), 0);
187 
188  _rm = rm;
189  _addr = addr;
190  }

200  /**
201  * \brief Reserve the given area in the region map.
202  * \param start The virtual start address of the area to reserve.
203  * \param size The size of the area to reserve (in bytes).
204  * \param flags Flags for the reserved area (see #Region_flags and #Attach_flags).
205  * \param align Alignment of area if searched as bits (log2 value).
206  * \retval start Start of address.
207  * \return 0 on success, <0 on error
208  * - -#L4_EADDRNOTAVAIL
209  * - IPC errors
210  *
211  * This function reserves an area within the virtual address space implemented by
212  * the region map. There are two kinds of areas available:
213  * - Reserved areas (\a flags = #Reserved), where no data spaces can be attached
214  * - Special purpose areas (\a flags = 0), where data spaces can be attached to
215  * the area via the #In_area flag and a start address within the area itself.
216  *
217  * \note When searching for a free place in the virtual address space
218  * (with \a flags = Search_addr),
219  * the space between \a start and the end of the virtual address space is
220  * searched.
221  */
222  long reserve_area(l4_addr_t *start, unsigned long size,
223  unsigned flags = 0,
224  unsigned char align = L4_PAGESHIFT) const throw();

260  /**
261  * \brief Attach a data space to a region.
262  *
263  * \param start Virtual start address
264  * \param size Size of the data space to attach (in bytes)
265  * \param flags Flags, see #Attach_flags and #Region_flags
266  * \param mem Data space
267  * \param offs Offset into the data space to use
268  * \param align Alignment of the virtual region, log2-size, default:
269  * a page (#L4_PAGESHIFT),
270  * Only meaningful if the #Search_addr flag is used.
271  * \retval start Start of region if #Search_addr was used.
272  *
273  * \return 0 on success, <0 on error
274  * - -#L4_ENOENT
275  * - -#L4_EPERM
276  * - -#L4_EINVAL
277  * - -#L4_EADDRNOTAVAIL
278  * - IPC errors
279  *
280  * Makes the whole or parts of a data space visible in the virtual memory
281  * of the corresponding task. The corresponding region in the virtual
282  * address space is backed with the contents of the dataspace.
283  *
284  * \note When searching for a free place in the virtual address space,
285  * the space between \a start and the end of the virtual address space is
286  * searched.
287  *
288  * \note There is no region object created, instead the region is
289  * defined by a virtual address within this range (see #find).
290  */
291  long attach(l4_addr_t *start, unsigned long size, unsigned long flags,
292  L4::Cap<Dataspace> mem, l4_addr_t offs = 0,
293  unsigned char align = L4_PAGESHIFT) const throw();

295  /**
296  * \brief Attach a dataspace to a region.
297  *
298  * \see attach
299  */
300  template< typename T >
301  long attach(T **start, unsigned long size, unsigned long flags,
302  L4::Cap<Dataspace> mem, l4_addr_t offs = 0,
303  unsigned char align = L4_PAGESHIFT) const throw()
304  {
305  union X { l4_addr_t a; T* t; };
306  X *x = reinterpret_cast<X*>(start);
307  return attach(&x->a, size, flags, mem, offs, align);
308  }

310  template< typename T >
311  long attach(Auto_region<T> *start, unsigned long size, unsigned long flags,
312  L4::Cap<Dataspace> mem, l4_addr_t offs = 0,
313  unsigned char align = L4_PAGESHIFT) const throw()
314  {
315  l4_addr_t addr = (l4_addr_t)start->get();
316 
317  long res = attach(&addr, size, flags, mem, offs, align);
318  if (res < 0)
319  return res;
320 
321  start->reset((T)addr, L4::Cap<Rm>(cap()));
322  return res;
323  }

384  /**
385  * \brief Find a region given an address and size.
386  *
387  * \param addr Address to look for
388  * \param size Size of the area to look for (in bytes).
389  * \retval addr Start address of the found region.
390  * \retval size Size of the found region (in bytes).
391  * \retval offset Offset at the beginning of the region within the
392  * associated dataspace.
393  * \retval flags Region flags, see #Region_flags.
394  * \retval m Associated dataspace or paging service.
395  *

399  * - IPC errors

402  * described by the addr and size parameter.
403  *
404  * \verbatim
405  size-out
406  / \
407  / \
408  addr-out \
409  ^________________\
410  ------------|----------------|------------------
411  | | Region | Dataspace |
412  | |_______|___|____| |
413  ------------|-------|---|-----------------------
414  \ / |\ /
415  \ / | |> size-in
416  offset-out |
417  |> addr-in

419  *
420  *
421  * \note The value of the size input parameter should be 1 to assure that a
422  * region can be determined unambiguously.
423  *
