41 namespace L4
42 {
43  namespace Kip
44  {
45  /**
46  * \brief Memory descriptors stored in the kernel interface page.
47  * \ingroup l4_kip_api
48  *
49  * <c>\#include <l4/sys/kip></c>
50  */
51  class Mem_desc
52  {
53  public:
54  /**
55  * \brief Memory types.
56  */
57  enum Mem_type
58  {
59  Undefined = 0x0, //< Undefined memory
60  Conventional = 0x1, //< Conventional memory
61  Reserved = 0x2, //< Reserved region, do not use this memory
62  Dedicated = 0x3, //< Dedicated
63  Shared = 0x4, //< Shared
64 
65  Info = 0xd, //< Info by boot loader
66  Bootloader = 0xe, //< Memory belongs to the boot loader
67  Arch = 0xf //< Architecture specific memory
68  };
69  private:
70  unsigned long _l, _h;
71 
72  static unsigned long &memory_info(void *kip) throw()
73  { return *((unsigned long *)kip + 21); }
74 
75  static unsigned long memory_info(void const *kip) throw()
76  { return *((unsigned long const *)kip + 21); }
77 
78  public:
79  /**
80  * \brief Get first memory descriptor.
81  * \param kip Pointer to the kernel info page
82  * \return First memory descriptor stored in the kernel info page
83  */
84  static Mem_desc *first(void *kip) throw()
85  {
86  return (Mem_desc *)((char *)kip
87  + (memory_info(kip) >> ((sizeof(unsigned long) / 2) * 8)));
88  }
89 
90  static Mem_desc const *first(void const *kip) throw()
91  {
92  return (Mem_desc const *)((char const *)kip
93  + (memory_info(kip) >> ((sizeof(unsigned long) / 2) * 8)));
94  }
95 
96  /**
97  * \brief Return number of memory descriptors stored in the kernel info page.
98  * \param kip Pointer to the kernel info page
99  * \return Number of memory descriptors in the kernel info page.
100  */
101  static unsigned long count(void const *kip) throw()
102  {
103  return memory_info(kip)
104  & ((1UL << ((sizeof(unsigned long) / 2) * 8)) - 1);
105  }
106 
107  /**
108  * \brief Set number of memory descriptors.
109  * \param kip Pointer to the kernel info page
110  * \param count Number of memory descriptors
111  */
112  static void count(void *kip, unsigned count) throw()
113  {
114  unsigned long &mi = memory_info(kip);
115  mi = (mi & ~((1UL << ((sizeof(unsigned long) / 2) * 8)) - 1)) | count;
116  }
117 
118  /**
119  * \brief Initialize memory descriptor.
120  * \param start Start address
121  * \param end End address
122  * \param t Memory type
123  * \param st Memory subtype, defaults to 0
124  * \param virt True for virtual memory, false for physical memory,
125  * defaults to physical
126  */
127  Mem_desc(unsigned long start, unsigned long end,
128  Mem_type t, unsigned char st = 0, bool virt = false) throw()
129  : _l((start & ~0x3ffUL) | (t & 0x0f) | ((st << 4) & 0x0f0)
130  | (virt ? 0x0200 : 0x0)), _h(end | 0x3ffUL)
131  {}
132 
133  /**
134  * \brief Return start address of memory descriptor.
135  * \return Start address of memory descriptor
136  */
137  unsigned long start() const throw() { return _l & ~0x3ffUL; }
138 
139  /**
140  * \brief Return end address of memory descriptor.
141  * \return End address of memory descriptor
142  */
143  unsigned long end() const throw() { return _h | 0x3ffUL; }
144 
145  /**
146  * \brief Return size of region described by the memory descriptor.
147  * \return Size of the region described by the memory descriptor
148  */
149  unsigned long size() const throw() { return end() + 1 - start(); }
150 
151  /**
152  * \brief Return type of the memory descriptor.
153  * \return Type of the memory descriptor
154  */
155  Mem_type type() const throw() { return (Mem_type)(_l & 0x0f); }
156 
157  /**
158  * \brief Return sub-type of the memory descriptor.
159  * \return Sub-type of the memory descriptor
160  */
161  unsigned char sub_type() const throw() { return (_l >> 4) & 0x0f; }
162 
163  /**
164  * \brief Return whether the memory descriptor describes a virtual or
165  * physical region.
166  * \return True for virtual region, false for physical region.
167  */
168  unsigned is_virtual() const throw() { return _l & 0x200; }
169 
170  /**
171  * \brief Set values of a memory descriptor.
172  * \param start Start address
173  * \param end End address
174  * \param t Memory type
175  * \param st Sub-type, defaults to 0
176  * \param virt Virtual or physical memory region, defaults to physical
177  */
178  void set(unsigned long start, unsigned long end,
179  Mem_type t, unsigned char st = 0, bool virt = false) throw()
180  {
181  _l = (start & ~0x3ffUL) | (t & 0x0f) | ((st << 4) & 0x0f0)
182  | (virt?0x0200:0x0);
183 
184  _h = end | 0x3ffUL;
185  }
186 
187  };
188  };
189 };
